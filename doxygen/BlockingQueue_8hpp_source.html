102  template <class Z>
103  void push (Z &&val) // forwarding reference
104  {
105  std::unique_lock<std::mutex> pushLock (m_pushLock); // only one lock on this branch
106  if (!pushElements.empty ())
107  {
108  pushElements.push_back (std::forward<Z> (val));
109  }
110  else
111  {
112  bool expEmpty = true;
113  if (queueEmptyFlag.compare_exchange_strong (expEmpty, false))
114  {
115  // release the push lock so we don't get a potential deadlock condition
116  pushLock.unlock ();
117  std::unique_lock<std::mutex> pullLock (m_pullLock); // first pullLock
118  queueEmptyFlag = false;
119  if (pullElements.empty ())
120  {
121  pullElements.push_back (std::forward<Z> (val));
122  // pullLock.unlock ();
123  condition.notify_all ();
124  }
125  else
126  {
127  pushLock.lock ();
128  pushElements.push_back (std::forward<Z> (val));
129  }
130  }

200  T pop ()
201  {
202  auto val = try_pop ();
203  while (!val)
204  {
205  std::unique_lock<std::mutex> pullLock (m_pullLock); // get the lock then wait
206  if (!pullElements.empty ()) // make sure we are actually empty;
207  {
208  auto actval = std::move (pullElements.back ());
209  pullElements.pop_back ();
210  return actval;
211  }
212  condition.wait (pullLock); // now wait
213  if (!pullElements.empty ()) // check for spurious wake-ups
214  {
215  auto actval = std::move (pullElements.back ());
216  pullElements.pop_back ();
217  return actval;
218  }
219  pullLock.unlock ();
220  val = try_pop ();
221  }
222  // move the value out of the optional
223  return std::move (*val);
224  }

279  if (!pushElements.empty ())
280  { // on the off chance the queue got out of sync
281  std::swap (pushElements, pullElements);
282  pushLock.unlock (); // we can free the push function to accept more elements after the swap call;
283  std::reverse (pullElements.begin (), pullElements.end ());
284  stx::optional<T> val (std::move (pullElements.back ())); // do it this way to allow movable only types
285  pullElements.pop_back ();
286  if (pullElements.empty ())
287  {
288  pushLock.lock (); // second pushLock

307  pullElements.pop_back ();
