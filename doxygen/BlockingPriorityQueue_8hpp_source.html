107  if (!pushElements.empty ())
108  {
109  pushElements.push_back (std::forward<Z> (val));
110  }
111  else
112  {
113  bool expEmpty = true;
114  if (queueEmptyFlag.compare_exchange_strong (expEmpty, false))
115  {
116  // release the push lock so we don't get a potential deadlock condition
117  pushLock.unlock ();

119  queueEmptyFlag = false;
120  if (pullElements.empty ())
121  {
122  pullElements.push_back (std::forward<Z> (val));
123  // pullLock.unlock ();

268  {

270  while (!val)
271  {
272  std::unique_lock<std::mutex> pullLock (m_pullLock); // get the lock then wait
273  if (!priorityQueue.empty ())
274  {
275  auto actval = std::move (priorityQueue.front ());
276  priorityQueue.pop ();
277  return actval;
278  }
279  if (!pullElements.empty ()) // make sure we are actually empty;
280  {
281  auto actval = std::move (pullElements.back ());
282  pullElements.pop_back ();
283  return actval;
284  }
285  condition.wait (pullLock); // now wait
286  if (!priorityQueue.empty ())
287  {
288  auto actval = std::move (priorityQueue.front ());
289  priorityQueue.pop ();
290  return actval;
291  }
292  if (!pullElements.empty ()) // check for spurious wake-ups
293  {
294  auto actval = std::move (pullElements.back ());
295  pullElements.pop_back ();
296  return actval;
297  }

361 {
362  std::lock_guard<std::mutex> pullLock (m_pullLock); // first pullLock
363  if (!priorityQueue.empty ())
364  {
365  stx::optional<T> val (std::move (priorityQueue.front ()));
366  priorityQueue.pop ();
367  return val;
368  }
369  if (pullElements.empty ())
370  {
371  std::unique_lock<std::mutex> pushLock (m_pushLock); // second pushLock
372  if (!pushElements.empty ())
373  { // on the off chance the queue got out of sync
374  std::swap (pushElements, pullElements);
375  pushLock.unlock (); // we can free the push function to accept more elements after the swap call;
376  std::reverse (pullElements.begin (), pullElements.end ());
377  stx::optional<T> val (std::move (pullElements.back ())); // do it this way to allow movable only types
378  pullElements.pop_back ();
379  if (pullElements.empty ())
380  {
381  pushLock.lock (); // second pushLock
382  if (!pushElements.empty ()) // more elements could have been added
383  { // this is the potential for slow operations
384  std::swap (pushElements, pullElements);
385  // we can free the push function to accept more elements after the swap call;
386  pushLock.unlock ();
387  std::reverse (pullElements.begin (), pullElements.end ());
388  }
389  else
390  {
391  queueEmptyFlag = true;
392  }
393  }
394  return val;
395  }
396  queueEmptyFlag = true;
397  return {}; // return the empty optional
398  }
399  stx::optional<T> val (std::move (pullElements.back ())); // do it this way to allow movable only types
400  pullElements.pop_back ();
401  if (pullElements.empty ())
402  {
403  std::unique_lock<std::mutex> pushLock (m_pushLock); // second pushLock
404  if (!pushElements.empty ())
