46 class shared_queue_state
47 {
48  private:
49  using ipcmutex = boost::interprocess::interprocess_mutex;
50  mutable ipcmutex data_lock;
51  queue_state_t state = queue_state_t::startup;
52 
53  public:
54  queue_state_t getState () const
55  {
56  try
57  {
58  boost::interprocess::scoped_lock<ipcmutex> lock (data_lock);
59  return state;
60  }
61  catch (const boost::interprocess::lock_exception &)
62  {
63  return queue_state_t::unknown;
64  }
65  }
66  bool setState (queue_state_t newState)
67  {
68  bool success = false;
69  int tries = 0;
70  while (!success)
71  {
72  try
73  {
74  boost::interprocess::scoped_lock<ipcmutex> lock (data_lock);
75  state = newState;
76  success = true;
77  }
78  catch (const boost::interprocess::lock_exception &)
79  {
80  std::this_thread::sleep_for (std::chrono::milliseconds (100));
81  ++tries;
82  if (tries > 20)
83  {
84  std::cout << "error in connecting to process lock\n";
85  state = newState;
86  return false;
87  }
88  }
89  }
90  return success;
91  }
92 };
