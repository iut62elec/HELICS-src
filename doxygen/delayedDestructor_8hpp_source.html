69  size_t destroyObjects ()
70  {
71  std::lock_guard<std::mutex> lock (destructionLock);
72  if (!ElementsToBeDestroyed.empty ())
73  {
74  if (callBeforeDeleteFunction)
75  {
76  for (auto &element : ElementsToBeDestroyed)
77  {
78  if (element.use_count () == 1)
79  {
80  callBeforeDeleteFunction (element);
81  }
82  }
83  }
84  // so apparently remove_if can actually call the destructor for shared_ptrs so the call function needs
85  // to be before this call
86  auto loc = std::remove_if (ElementsToBeDestroyed.begin (), ElementsToBeDestroyed.end (),
87  [](const auto &element) { return (element.use_count () <= 1); });
88  ElementsToBeDestroyed.erase (loc, ElementsToBeDestroyed.end ());
89  }
90  return ElementsToBeDestroyed.size ();
91  }

93  size_t destroyObjects (int delay)
94  {
95  std::unique_lock<std::mutex> lock (destructionLock);
96  auto delayTime = std::chrono::milliseconds ((delay < 100) ? delay : 50);
97  int delayCount = (delay < 100) ? 1 : (delay / 50);
98 
99  int cnt = 0;
100  while ((!ElementsToBeDestroyed.empty ()) && (cnt < delayCount))
101  {
102  if (cnt > 0)
103  {
104  lock.unlock ();
105  std::this_thread::sleep_for (delayTime);
106  ++cnt;
107  lock.lock ();
108  }
109  else
110  {
111  ++cnt;
112  }
113 
114  if (!ElementsToBeDestroyed.empty ())
115  {
116  auto loc = std::remove_if (ElementsToBeDestroyed.begin (), ElementsToBeDestroyed.end (),
117  [](const auto &element) { return (element.use_count () <= 1); });
118  if (callBeforeDeleteFunction)
119  {
120  auto locIt = loc;
121  while (locIt != ElementsToBeDestroyed.end ())
122  {
123  if (*locIt)
124  {
125  callBeforeDeleteFunction (*locIt);
126  }
127  ++locIt;
128  }
129  }
130  ElementsToBeDestroyed.erase (loc, ElementsToBeDestroyed.end ());
131  }
132  }
133  return ElementsToBeDestroyed.size ();
