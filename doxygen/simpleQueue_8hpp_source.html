112  void push (Z &&val) // forwarding reference
113  {
114  std::unique_lock<std::mutex> pushLock (m_pushLock); // only one lock on this branch
115  if (!pushElements.empty ())
116  {
117  pushElements.push_back (std::forward<Z> (val));
118  }
119  else
120  {
121  bool expEmpty = true;
122  if (queueEmptyFlag.compare_exchange_strong (expEmpty, false))
123  {
124  // release the push lock
125  pushLock.unlock ();
126  std::unique_lock<std::mutex> pullLock (m_pullLock); // first pullLock
127  queueEmptyFlag = false;
128  if (pullElements.empty ())
129  {
130  pullElements.push_back (std::forward<Z> (val));
131  pullLock.unlock ();
132  }
133  else
134  {
135  pushLock.lock ();
136  pushElements.push_back (std::forward<Z> (val));
137  }
138  }

196  if (!pushElements.empty ())
197  { // on the off chance the queue got out of sync
198  std::swap (pushElements, pullElements);
199  pushLock.unlock (); // we can free the push function to accept more elements after the swap call;
200  std::reverse (pullElements.begin (), pullElements.end ());
201  stx::optional<X> val (
202  std::move (pullElements.back ())); // do it this way to allow moveable only types
203  pullElements.pop_back ();
204  if (pullElements.empty ())
205  {
206  pushLock.lock (); // second pushLock
207  if (!pushElements.empty ()) // more elements could have been added
208  { // this is the potential for slow operations

211  pushLock.unlock ();

213  }

217  }

220  }
